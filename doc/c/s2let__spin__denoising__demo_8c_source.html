00029 void hard_threshold(
00030     complex double *g_wav,
00031     const double *threshold,
00032     const s2let_parameters_t *parameters
00033 ) {
00034     int L = parameters->L;
00035     int J_min = parameters->J_min;
00036     int N = parameters->N;
00037 
00038     int J = s2let_j_max(parameters);
00039     int i, j, offset = 0;
00040     for(j = J_min; j <= J; j++){
00041         int bl = parameters->upsample ? L : MIN(s2let_bandlimit(j, parameters), L);
00042         for(i = 0; i < N*bl*(2*bl-1); i++){
00043             if( cabs(g_wav[offset + i]) < threshold[j-J_min] )
00044                 g_wav[offset + i] = 0;
00045         }
00046         offset += N*bl*(2*bl-1);
00047     }
00048 }
